project(CemuComponents)

file(GLOB_RECURSE H_FILES *.h)

target_include_directories(CemuComponents PRIVATE ../)

if(ENABLE_DISCORD_RPC)

endif()
target_link_libraries(CemuComponents PRIVATE CemuUtil)
